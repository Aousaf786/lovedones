CREATE TABLE memorial_profile_images (
    id BigInt(20) NOT NULL AUTO_INCREMENT,
    memorial_profile_id BigInt(20) NOT NULL,
    img varchar(255),
    profile_img tinyint(1) DEFAULT '0',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    PRIMARY KEY (ID),
    CONSTRAINT memorial_profile_images_id_foreign FOREIGN KEY (memorial_profile_id) REFERENCES memorial_profiles(id)
);
